def bmeshGetElementoActivo(bm):
    active = bm.select_history.active
    if active is not None:
        return active
    for f in bm.faces:
        if f.select:
            return f
    for e in bm.edges:
        if e.select:
            return e
    for v in bm.verts:
        if v.select:
            return v
    return None

def mallaGetElementosSeleccionados(bm, modo):
    if modo == 'VERT':
        return [v for v in bm.verts if v.select]
    elif modo == 'EDGE':
        return [e for e in bm.edges if e.select]
    elif modo == 'FACE':
        return [f for f in bm.faces if f.select]
    return []


def elementGetActivoYOtros(bm, modo):
    active = bmeshGetElementoActivo(bm)

    if modo == 'VERT' and not isinstance(active, bmesh.types.BMVert):
        active = None
    elif modo == 'EDGE' and not isinstance(active, bmesh.types.BMEdge):
        active = None
    elif modo == 'FACE' and not isinstance(active, bmesh.types.BMFace):
        active = None

    seleccionados = mallaGetElementosSeleccionados(bm, modo)

    if active is None and seleccionados:
        active = seleccionados[0]

    otros = [e for e in seleccionados if e != active]

    return active, otros


def sceneGetAllselects(context):
    current_mode = context.mode

    if current_mode == 'OBJECT':
        active_obj = context.active_object
        selected_objects = [obj for obj in context.selected_objects if obj != active_obj]
        return active_obj, selected_objects

    elif current_mode == 'EDIT_MESH':
        obj = context.active_object
        if not obj or obj.type != 'MESH':
            print("Error: No hay objeto de malla activo en Modo Edición.")
            return None, []

        bm = bmesh.from_edit_mesh(obj.data)
        bm.select_history.validate()

        select_mode = context.tool_settings.mesh_select_mode[:]

        active_element = None
        other_selected_elements = []

        if select_mode[0]:
            active_element = bm.select_history.active
            other_selected_elements = [v for v in bm.verts if v.select and v != active_element]
        elif select_mode[1]:
            active_element = bm.select_history.active
            other_selected_elements = [e for e in bm.edges if e.select and e != active_element]
        elif select_mode[2]:
            active_element = bm.select_history.active
            other_selected_elements = [f for f in bm.faces if f.select and f != active_element]
        else:
            print("Ningún modo de selección de malla (vértices, aristas, caras) está activo.")

        bmesh.update_edit_mesh(obj.data)
        bm.free()

        return active_element, other_selected_elements

    else:
        print(f"Modo no soportado o sin selección: {current_mode}")
        return None, []

def elementGetCentroLocal(element):
    if isinstance(element, bmesh.types.BMVert):
        return element.co.copy()
    elif isinstance(element, bmesh.types.BMEdge):
        v1, v2 = element.verts
        return (v1.co + v2.co) / 2.0
    elif isinstance(element, bmesh.types.BMFace):
        return element.calc_center_median()
    return None


def elementGetCentroGlobal(element, matrix_world):
    if hasattr(element, 'co'):
        return matrix_world @ element.co
    avg = sum((v.co for v in element.verts), Vector()) / len(element.verts)
    return matrix_world @ avg


def elementGetVertices(element):
    if isinstance(element, bmesh.types.BMVert):
        return [element]
    elif isinstance(element, bmesh.types.BMEdge):
        return list(element.verts)
    elif isinstance(element, bmesh.types.BMFace):
        return list(element.verts)
    return []


def elementGetNormalGlobal(element, matrix_world):
    mat3 = matrix_world.to_3x3()

    if isinstance(element, bmesh.types.BMVert):
        normal = (mat3 @ element.normal).normalized()

    elif isinstance(element, bmesh.types.BMFace):
        normal = (mat3 @ element.normal).normalized()

    elif isinstance(element, bmesh.types.BMEdge):
        v1, v2 = element.verts
        linked_faces = element.link_faces
        if linked_faces:
            avg_normal = Vector((0, 0, 0))
            for f in linked_faces:
                avg_normal += f.normal
            avg_normal /= len(linked_faces)
            normal = (mat3 @ avg_normal).normalized()
        else:
            direction = ((matrix_world @ v2.co) - (matrix_world @ v1.co)).normalized()
            if abs(direction.z) < 0.9:
                normal = direction.cross(Vector((0, 0, 1))).normalized()
            else:
                normal = direction.cross(Vector((1, 0, 0))).normalized()
    else:
        return Vector((0, 0, 1))

    if normal.length < 0.001:
        normal = Vector((0, 0, 1))
    return normal


def objectoGetMatrizOrientacion(context, obj, bm, orientacion):
    if orientacion == 'GLOBAL':
        return Matrix.Identity(3)
    elif orientacion == 'LOCAL':
        return obj.matrix_world.to_3x3()
    elif orientacion == 'CURSOR':
        return context.scene.cursor.matrix.to_3x3()
    elif orientacion == 'VIEW':
        if context.space_data and hasattr(context.space_data, 'region_3d'):
            return context.space_data.region_3d.view_matrix.inverted().to_3x3()
        return Matrix.Identity(3)
    elif orientacion == 'NORMAL':
        active = bm.select_history.active
        if active:
            n = active.normal.copy()
            if abs(n.x) < 0.9:
                t = Vector((1, 0, 0))
            else:
                t = Vector((0, 1, 0))
            b = n.cross(t).normalized()
            t2 = b.cross(n).normalized()
            return Matrix((t2, b, n)).transposed()
        return Matrix.Identity(3)
    elif orientacion == 'GIMBAL':
        return obj.rotation_euler.to_matrix()
    return Matrix.Identity(3)


def meshGetEdgesSeleccionadasGlobal(obj):
    if not obj or obj.type != 'MESH' or obj.mode != 'EDIT':
        return []
    bm = bmesh.from_edit_mesh(obj.data)
    bm.edges.ensure_lookup_table()
    mw = obj.matrix_world
    resultado = []
    for edge in bm.edges:
        if edge.select:
            p1 = mw @ edge.verts[0].co
            p2 = mw @ edge.verts[1].co
            resultado.append((p1, p2))
    bmesh.update_edit_mesh(obj.data)
    bm.free()
    return resultado


def mallaMirrorGeometria(obj, bm, mirror_x, mirror_y, mirror_z, orient_mat):
    geom_to_dup = []
    for v in bm.verts:
        if v.select:
            geom_to_dup.append(v)
    for e in bm.edges:
        if e.select:
            geom_to_dup.append(e)
    for f in bm.faces:
        if f.select:
            geom_to_dup.append(f)

    if not geom_to_dup:
        return None

    for v in bm.verts:
        v.select = False
    for e in bm.edges:
        e.select = False
    for f in bm.faces:
        f.select = False

    ret = bmesh.ops.duplicate(bm, geom=geom_to_dup)
    new_geom = ret['geom']
    new_verts = [g for g in new_geom if isinstance(g, bmesh.types.BMVert)]

    if not new_verts:
        return None

    for v in new_verts:
        v.select = True

    centroid = Vector((0, 0, 0))
    for v in new_verts:
        centroid += v.co
    centroid /= len(new_verts)

    orient_mat_inv = orient_mat.inverted()
    centroid_custom = orient_mat_inv @ centroid

    if mirror_x:
        centroid_custom.x *= -1.0
    if mirror_y:
        centroid_custom.y *= -1.0
    if mirror_z:
        centroid_custom.z *= -1.0

    new_centroid = orient_mat @ centroid_custom
    offset = new_centroid - centroid
    for v in new_verts:
        v.co += offset

    for g in new_geom:
        g.select = True

    return new_geom


def mallaSeleccionarPorRango(obj, bm, modo, vmin, vmax):
    for v in bm.verts:
        v.select = False
    for e in bm.edges:
        e.select = False
    for f in bm.faces:
        f.select = False

    count = 0
    if modo == 'EDGE':
        for e in bm.edges:
            v1, v2 = e.verts
            length = ((obj.matrix_world @ v1.co) - (obj.matrix_world @ v2.co)).length
            if vmin <= length <= vmax:
                e.select = True
                count += 1
    elif modo == 'FACE':
        for f in bm.faces:
            if vmin <= f.calc_area() <= vmax:
                f.select = True
                count += 1

    bm.select_flush_mode()
    return count


def objectoAjustarDistanciaObjeto(obj_active, obj_ref, target_dist):
    dir_vec = obj_active.location - obj_ref.location
    if dir_vec.length < 0.0001:
        return False
    obj_active.location = obj_ref.location + (dir_vec.normalized() * target_dist)
    return True


def mallaAjustarDistanciaElemento(obj, bm, modo, target_dist):
    elem_active, otros = elementGetActivoYOtros(bm, modo)
    elem_ref = otros[0] if otros else None

    if elem_active is None or elem_ref is None:
        return False

    mat_world = obj.matrix_world
    center_active_global = elementGetCentroGlobal(elem_active, mat_world)
    center_ref_global = elementGetCentroGlobal(elem_ref, mat_world)
    dir_vec_global = center_active_global - center_ref_global
    if dir_vec_global.length < 0.0001:
        return False

    new_center_active_global = center_ref_global + (dir_vec_global.normalized() * target_dist)
    offset_local = mat_world.inverted().to_3x3() @ (new_center_active_global - center_active_global)
    for v in elementGetVertices(elem_active):
        v.co += offset_local

    return True


def mallaObtenerTargetsPorModo(obj, bm, modo):
    mat_world = obj.matrix_world
    targets = []

    if modo == 'VERT':
        for v in bm.verts:
            if v.select:
                targets.append((mat_world @ v.co, elementGetNormalGlobal(v, mat_world)))
    elif modo == 'EDGE':
        for e in bm.edges:
            if e.select:
                targets.append((elementGetCentroGlobal(e, mat_world), elementGetNormalGlobal(e, mat_world)))
    elif modo == 'FACE':
        for f in bm.faces:
            if f.select:
                targets.append((elementGetCentroGlobal(f, mat_world), elementGetNormalGlobal(f, mat_world)))

    return targets

def objectoalinearEjeADireccion(obj, direction, axis="Z"):
    axis = axis.upper()
    target = direction.normalized()

    if target.length < 1e-9:
        raise ValueError("El vector de direccion no puede ser cero.")

    aux = Vector((0, 1, 0))
    if abs(target.dot(aux)) > 0.99:
        aux = Vector((1, 0, 0))

    if axis == "Z":
        z = target
        x = z.cross(aux).normalized()
        y = z.cross(x).normalized()
    elif axis == "Y":
        y = target
        x = y.cross(aux).normalized()
        z = x.cross(y).normalized()
    elif axis == "X":
        x = target
        y = x.cross(aux).normalized()
        z = x.cross(y).normalized()
    else:
        raise ValueError(f"Eje '{axis}' no valido. Usa 'X', 'Y' o 'Z'.")

    rot = Matrix((
        (x.x, y.x, z.x),
        (x.y, y.y, z.y),
        (x.z, y.z, z.z),
    ))
    obj.rotation_euler = rot.to_euler('XYZ')


def objectoSetGlobalPostion(obj, x, y, z):
    obj.location = Vector((x, y, z))


def objectoSetDimencion(obj, value, axis="Z"):
    axis = axis.upper()
    idx = {"X": 0, "Y": 1, "Z": 2}[axis]

    current = obj.dimensions[idx]
    if current < 1e-9:
        raise ValueError(f"El objeto '{obj.name}' tiene dimension 0 en el eje {axis}.")

    factor = value / current
    scale = list(obj.scale)
    scale[idx] *= factor
    obj.scale = scale

    original_active = bpy.context.view_layer.objects.active
    original_selection = bpy.context.selected_objects[:]

    try:
        bpy.ops.object.select_all(action='DESELECT')
        obj.select_set(True)
        bpy.context.view_layer.objects.active = obj

        bpy.ops.object.transform_apply(location=False, rotation=False, scale=True)
    finally:
        bpy.ops.object.select_all(action='DESELECT')
        for o in original_selection:
            o.select_set(True)
        bpy.context.view_layer.objects.active = original_active


def objectoClonarYColocar(
    source,
    pos1,
    pos2,
    scale_x=False,
    scale_y=False,
    scale_z=True,
    align_axis="Z",
):
    direccion = pos2 - pos1
    distancia = direccion.length
    if distancia < 1e-6:
        return None

    clone = objectoDuplicarObjeto(source)
    if clone is None:
        return None

    clone.name = source.name + "_clone"

    if scale_x:
        objectoSetDimencion(clone, distancia, "X")
    if scale_y:
        objectoSetDimencion(clone, distancia, "Y")
    if scale_z:
        objectoSetDimencion(clone, distancia, "Z")

    objectoSetGlobalPostion(clone, pos1.x, pos1.y, pos1.z)

    objectoalinearEjeADireccion(clone, direccion, align_axis)

    return clone


def coleccionExiste(nombre):

def coleccionCrear(nombre, parent_collection=None):
    coleccion = bpy.data.collections.new(nombre)
    if parent_collection is None:
        parent_collection = bpy.context.scene.collection
    parent_collection.children.link(coleccion)
    return coleccion


def coleccionObtenerOCrear(nombre, parent_collection=None):
    coleccion = bpy.data.collections.get(nombre)
    if coleccion is None:
        coleccion = coleccionCrear(nombre, parent_collection)
    return coleccion


def objectoAñadirAColeccion(obj, coleccion, remover_de_otras=False):
    if remover_de_otras:
        for col in obj.users_collection:
            col.objects.unlink(obj)
    if obj.name not in coleccion.objects:
        coleccion.objects.link(obj)
    return obj


def objectoRemoverDeColeccion(obj, coleccion):
    if obj.name in coleccion.objects:
        coleccion.objects.unlink(obj)
    return obj


def coleccionGetObjetos(nombre):
    coleccion = bpy.data.collections.get(nombre)
    if coleccion is None:
        return []
    return list(coleccion.objects)


def coleccionEliminar(nombre, eliminar_objetos=False):
    coleccion = bpy.data.collections.get(nombre)
    if coleccion is None:
        return False
    if eliminar_objetos:
        for obj in list(coleccion.objects):
            bpy.data.objects.remove(obj, do_unlink=True)
    bpy.data.collections.remove(coleccion)
    return True


def objectoClonarEnPosiciones(
    valid_refs,
    targets,
    track_axis="Z",
    up_axis="Y",
    use_advanced=False,
    rot_ranges=None,
    scl_ranges=None,
    collection_name="creates",
):
    if rot_ranges is None:
        rot_ranges = {}
    if scl_ranges is None:
        scl_ranges = {}

    creates_collection = coleccionObtenerOCrear(collection_name)

    creados = []

    for i, (pos, normal_dir) in enumerate(targets):
        ref_obj = valid_refs[i % len(valid_refs)]

        new_obj = ref_obj.copy()
        new_obj.data = ref_obj.data.copy()
        objectoAñadirAColeccion(new_obj, creates_collection)

        align_quat = normal_dir.to_track_quat(track_axis, up_axis)

        if use_advanced:
            eul_local = Euler((0.0, 0.0, 0.0), 'XYZ')

            if rot_ranges.get('x'):
                eul_local.x = math.radians(random.uniform(*rot_ranges['x']))
            if rot_ranges.get('y'):
                eul_local.y = math.radians(random.uniform(*rot_ranges['y']))
            if rot_ranges.get('z'):
                eul_local.z = math.radians(random.uniform(*rot_ranges['z']))

            random_quat = eul_local.to_quaternion()
            final_quat = align_quat @ random_quat
        else:
            final_quat = align_quat

        new_obj.location = pos
        new_obj.rotation_mode = 'QUATERNION'
        new_obj.rotation_quaternion = final_quat

        new_obj.scale = ref_obj.scale.copy()

        if use_advanced:
            if scl_ranges.get('x'):
                new_obj.scale.x *= random.uniform(*scl_ranges['x'])
            if scl_ranges.get('y'):
                new_obj.scale.y *= random.uniform(*scl_ranges['y'])
            if scl_ranges.get('z'):
                new_obj.scale.z *= random.uniform(*scl_ranges['z'])

        new_obj.select_set(False)
        creados.append(new_obj)

    return creados


def sceneGetObjectProperties(obj):
    if obj.type != 'MESH':
        dimensions = Vector((0, 0, 0))
    else:
        dimensions = obj.dimensions
    pos_local = obj.location
    pos_global = obj.matrix_world.translation
    obj_rot_matrix = obj.matrix_world.to_3x3().normalized()
    vec_x_local_global = obj_rot_matrix @ Vector((1, 0, 0))
    vec_y_local_global = obj_rot_matrix @ Vector((0, 1, 0))
    vec_z_local_global = obj_rot_matrix @ Vector((0, 0, 1))
    return {
        'name': obj.name,
        'isActive': (obj == bpy.context.active_object),
        'dimensions': dimensions,
        'posLocal': pos_local,
        'posGlobal': pos_global,
        'x': vec_x_local_global,
        'y': vec_y_local_global,
        'z': vec_z_local_global,
    }


def sceneGetObjetosScene():

    return object_properties

def is_face_selected(face):
    return face.select


def get_face_properties(face, obj_matrix_world):
    face_index = face.index
    face_normal = face.normal
    face_area = face.calc_area()
    face_center_local = sum([v.co for v in face.verts], Vector()) / len(face.verts)
    face_center_global = obj_matrix_world @ face_center_local

    return {
        'area': face_area,
        'index': face_index,
        "posLocal": face_center_local,
        'posGlobal': face_center_global,
        'normal': face_normal,
    }


def facesIterateObjectFaces(obj):
    if not obj or obj.mode != 'EDIT':
        print("Object is not in edit mode or is not valid.")
        return []

    bm = bmesh.from_edit_mesh(obj.data)
    selected_properties = []
    obj_matrix_world = obj.matrix_world

    for face in bm.faces:
        if is_face_selected(face):
            properties = get_face_properties(face, obj_matrix_world)
            selected_properties.append(properties)

def anguloNormalVsEje(normal, eje=None):
    if eje is None:
        eje = Vector((0.0, 0.0, 1.0))
    n = normal.normalized()
    dot = max(-1.0, min(1.0, n.dot(eje)))
    angle = math.degrees(math.acos(dot))
    if angle < 0:
        angle = angle * -1.0
    return angle


def anguloNormalMundoVsEje(normal_local, matrix_world, eje=None):
    if eje is None:
        eje = Vector((0.0, 0.0, 1.0))
    normal_mat = matrix_world.to_3x3().inverted().transposed()
    world_normal = (normal_mat @ normal_local).normalized()
    dot = max(-1.0, min(1.0, world_normal.dot(eje)))
    return math.degrees(math.acos(dot))


def pesoLinealZ(z, z_min, z_max, val_min=0.4, val_max=1.0):
    denom = (z_max - z_min) if (z_max - z_min) != 0 else 1.0
    t = max(0.0, min(1.0, (z - z_min) / denom))
    return val_min + t * (val_max - val_min)


def pesoSimetricoZ(z, z_mid, z_max, val_min=0.4, val_max=1.0, invertido=False):
    denom = (z_max - z_mid) if (z_max - z_mid) != 0 else 1.0
    t = max(0.0, min(1.0, abs(z - z_mid) / denom))
    if invertido:
        return val_max - t * (val_max - val_min)
    return val_min + t * (val_max - val_min)

from mathutils import Vector, Matrix


def utilRotarPuntosAlrededorEje(points, axis_start, axis_direction, angle_degrees):
    angle_rad = math.radians(angle_degrees)
    axis_normalized = axis_direction.normalized()
    rotation_matrix = Matrix.Rotation(angle_rad, 4, axis_normalized)
    rotated_points = []
    for point in points:
        relative_point = point - axis_start
        rotated_relative = rotation_matrix @ relative_point
        rotated_points.append(rotated_relative + axis_start)
    return rotated_points


def mallaCrearArco(p1, p2, arc_type='CIRCULAR', height_factor=0.3, num_verts=20, curvature=1.0, rotation_angle=0.0, nombre="ArcoExtendido"):
    mesh = bpy.data.meshes.new(nombre)
    obj = bpy.data.objects.new(nombre, mesh)
    bpy.context.collection.objects.link(obj)

    direction_vector = p2 - p1
    distance = direction_vector.length
    if distance == 0:
        raise ValueError("Los puntos no pueden ser identicos")

    direction_normalized = direction_vector.normalized()
    up_vector = Vector((0, 0, 1))
    if abs(direction_normalized.dot(up_vector)) > 0.9:
        up_vector = Vector((1, 0, 0))
    perpendicular = direction_normalized.cross(up_vector).normalized()
    arc_height = distance * height_factor
    center_point = (p1 + p2) * 0.5
    verts = []

    if arc_type == 'CIRCULAR':
        if arc_height <= 0:
            arc_height = distance * 0.1
        radius = (distance * distance + 4 * arc_height * arc_height) / (8 * arc_height)
        arc_center = center_point + perpendicular * (radius - arc_height)
        vec_to_p1 = p1 - arc_center
        vec_to_p2 = p2 - arc_center
        angle_start = math.atan2(vec_to_p1.dot(perpendicular), vec_to_p1.dot(direction_normalized))
        angle_end = math.atan2(vec_to_p2.dot(perpendicular), vec_to_p2.dot(direction_normalized))
        if angle_end < angle_start:
            angle_end += 2 * math.pi
        for i in range(num_verts):
            t = i / (num_verts - 1)
            angle = angle_start + t * (angle_end - angle_start)
            point = arc_center + radius * (math.cos(angle) * direction_normalized + math.sin(angle) * perpendicular)
            verts.append(point)
    elif arc_type == 'PARABOLIC':
        for i in range(num_verts):
            t = i / (num_verts - 1)
            linear_point = p1 + t * direction_vector
            parabolic_height = 4 * arc_height * t * (1 - t)
            point = linear_point + parabolic_height * perpendicular
            verts.append(point)
    elif arc_type == 'PARABOLIC_2':
        for i in range(num_verts):
            t = i / (num_verts - 1)
            x = t * distance
            y = arc_height * (1 - (2 * t - 1) ** 2)
            point = p1 + x * direction_normalized + y * perpendicular
            verts.append(point)
    elif arc_type == 'ELLIPTIC':
        semi_major = distance / 2
        semi_minor = arc_height
        ellipse_center = center_point
        for i in range(num_verts):
            t = i / (num_verts - 1)
            angle = math.pi * t
            x_ellipse = semi_major * math.cos(angle)
            y_ellipse = semi_minor * math.sin(angle)
            point = ellipse_center + x_ellipse * direction_normalized + y_ellipse * perpendicular
            verts.append(point)
    elif arc_type == 'ELLIPTIC_2':
        a = distance / 2
        b = arc_height
        center = center_point
        for i in range(num_verts):
            t = i / (num_verts - 1)
            angle = math.pi * t
            x = a * math.cos(angle)
            y = b * math.sin(angle)
            point = center + x * direction_normalized + y * perpendicular
            verts.append(point)
    elif arc_type == 'CATENARY':
        a = distance / (2 * math.sinh(arc_height / distance)) if arc_height > 0 else distance / 4
        for i in range(num_verts):
            t = i / (num_verts - 1)
            x = (t - 0.5) * distance
            y = a * (math.cosh(x / a) - 1)
            linear_point = p1 + t * direction_vector
            point = linear_point + y * perpendicular
            verts.append(point)
    elif arc_type == 'HYPERBOLIC':
        for i in range(num_verts):
            t = i / (num_verts - 1)
            x = (t - 0.5) * distance * 2
            y = arc_height * math.sqrt(1 + (x / distance) ** 2) - arc_height
            linear_point = p1 + t * direction_vector
            point = linear_point + y * perpendicular
            verts.append(point)
    elif arc_type == 'HYPERBOLIC_2':
        for i in range(num_verts):
            t = i / (num_verts - 1)
            x = t * distance
            y = arc_height / (1 + 10 * t) if t < 1 else 0
            point = p1 + x * direction_normalized + y * perpendicular
            verts.append(point)
    elif arc_type == 'CUBIC':
        for i in range(num_verts):
            t = i / (num_verts - 1)
            cubic_height = arc_height * (3 * t * t - 2 * t * t * t) * (1 - (2 * t - 1) ** 2)
            linear_point = p1 + t * direction_vector
            point = linear_point + cubic_height * perpendicular
            verts.append(point)
    elif arc_type == 'SINUSOIDAL':
        for i in range(num_verts):
            t = i / (num_verts - 1)
            sin_height = arc_height * math.sin(math.pi * t)
            linear_point = p1 + t * direction_vector
            point = linear_point + sin_height * perpendicular
            verts.append(point)
    elif arc_type == 'EXPONENTIAL':
        for i in range(num_verts):
            t = i / (num_verts - 1)
            exp_factor = curvature * 2
            if t <= 0.5:
                exp_height = arc_height * (math.exp(exp_factor * t) - 1) / (math.exp(exp_factor * 0.5) - 1)
            else:
                exp_height = arc_height * (math.exp(exp_factor * (1 - t)) - 1) / (math.exp(exp_factor * 0.5) - 1)
            linear_point = p1 + t * direction_vector
            point = linear_point + exp_height * perpendicular
            verts.append(point)
    elif arc_type == 'GOTHIC':
        for i in range(num_verts):
            t = i / (num_verts - 1)
            if t <= 0.5:
                angle = math.pi * t
                radius = distance / 2
                center = p1 + (distance / 4) * direction_normalized
                point = center + radius * (math.cos(angle) * direction_normalized + math.sin(angle) * perpendicular)
            else:
                angle = math.pi * (t - 0.5)
                radius = distance / 2
                center = p2 - (distance / 4) * direction_normalized
                point = center + radius * (math.cos(math.pi - angle) * direction_normalized + math.sin(math.pi - angle) * perpendicular)
            verts.append(point)
    elif arc_type == 'SPIRAL':
        for i in range(num_verts):
            t = i / (num_verts - 1)
            spiral_angle = curvature * math.pi * t
            spiral_radius = arc_height * (1 - t)
            spiral_x = spiral_radius * math.cos(spiral_angle)
            spiral_y = spiral_radius * math.sin(spiral_angle)
            linear_point = p1 + t * direction_vector
            binormal = direction_normalized.cross(perpendicular)
            point = linear_point + spiral_x * perpendicular + spiral_y * binormal
            verts.append(point)
    elif arc_type == 'BEZIER_QUAD':
        control_point = center_point + arc_height * perpendicular
        for i in range(num_verts):
            t = i / (num_verts - 1)
            point = (1 - t) ** 2 * p1 + 2 * (1 - t) * t * control_point + t ** 2 * p2
            verts.append(point)
    elif arc_type == 'BEZIER_CUBIC':
        control1 = p1 + (distance / 3) * direction_normalized + (arc_height * 0.5) * perpendicular
        control2 = p2 - (distance / 3) * direction_normalized + (arc_height * 0.5) * perpendicular
        for i in range(num_verts):
            t = i / (num_verts - 1)
            point = ((1 - t) ** 3 * p1 + 3 * (1 - t) ** 2 * t * control1 + 3 * (1 - t) * t ** 2 * control2 + t ** 3 * p2)
            verts.append(point)
    elif arc_type == 'BEZIER_CUBIC_2':
        control1 = p1 + (distance / 4) * direction_normalized + (arc_height * 0.8) * perpendicular
        control2 = p2 - (distance / 4) * direction_normalized + (arc_height * 0.8) * perpendicular
        for i in range(num_verts):
            t = i / (num_verts - 1)
            point = ((1 - t) ** 3 * p1 + 3 * (1 - t) ** 2 * t * control1 + 3 * (1 - t) * t ** 2 * control2 + t ** 3 * p2)
            verts.append(point)
    elif arc_type == 'LINEAR':
        for i in range(num_verts):
            t = i / (num_verts - 1)
            point = p1 + t * direction_vector
            verts.append(point)

    if abs(rotation_angle) > 0.001:
        verts = utilRotarPuntosAlrededorEje(verts, p1, direction_vector, rotation_angle)

    edges = [(i, i + 1) for i in range(len(verts) - 1)]
    mesh.from_pydata(verts, edges, [])
    mesh.update()
    return obj


def curvaCoeficienteBinomial(n, k):
    if k > n or k < 0:
        return 0
    if k == 0 or k == n:
        return 1
    result = 1
    for i in range(min(k, n - k)):
        result = result * (n - i) // (i + 1)
    return result


def curvaEvaluarBezier(control_points, t):
    n = len(control_points)
    if n == 1:
        return control_points[0]
    points = [Vector(p) for p in control_points]
    for r in range(1, n):
        for i in range(n - r):
            points[i] = (1 - t) * points[i] + t * points[i + 1]
    return points[0]


def curvaEvaluarBezierBernstein(control_points, t):
    n = len(control_points) - 1
    point = Vector((0, 0, 0))
    for i in range(n + 1):
        bernstein = curvaCoeficienteBinomial(n, i) * (t ** i) * ((1 - t) ** (n - i))
        point += bernstein * Vector(control_points[i])
    return point


def curvaBezierMultiPunto(control_points, num_points):
    if len(control_points) < 2:
        raise ValueError("Minimo 2 puntos de control")
    points = []
    for i in range(num_points):
        t = i / (num_points - 1)
        points.append(curvaEvaluarBezier(control_points, t))
    return points


def curvaRegresionPolinomial3D(points, degree=3):
    if len(points) < degree + 1:
        degree = max(1, len(points) - 1)
    points_array = np.array([(p[0], p[1], p[2]) for p in points])
    t_values = [0]
    for i in range(1, len(points)):
        dist = np.linalg.norm(points_array[i] - points_array[i - 1])
        t_values.append(t_values[-1] + dist)
    t_max = t_values[-1]
    if t_max > 0:
        t_values = [t / t_max for t in t_values]
    t_array = np.array(t_values)
    coeffs_x = np.polyfit(t_array, points_array[:, 0], degree)
    coeffs_y = np.polyfit(t_array, points_array[:, 1], degree)
    coeffs_z = np.polyfit(t_array, points_array[:, 2], degree)
    return coeffs_x, coeffs_y, coeffs_z


def curvaEvaluarPolinomio3D(coeffs_x, coeffs_y, coeffs_z, t):
    x = np.polyval(coeffs_x, t)
    y = np.polyval(coeffs_y, t)
    z = np.polyval(coeffs_z, t)
    return Vector((x, y, z))


def mallaCrearCurvaPolinomial(control_points, num_verts=20, degree=3, curve_type='polynomial', rotation_angle=0.0):
    if len(control_points) < 2:
        raise ValueError("Minimo 2 puntos")

    axis_direction = control_points[-1] - control_points[0]
    axis_start = control_points[0]

    if curve_type == 'bezier':
        curve_points = curvaBezierMultiPunto(control_points, num_verts)
    elif curve_type == 'bezier_bernstein':
        curve_points = []
        for i in range(num_verts):
            t = i / (num_verts - 1)
            curve_points.append(curvaEvaluarBezierBernstein(control_points, t))
    else:
        max_degree = len(control_points) - 1
        if degree > max_degree:
            degree = max_degree
        coeffs_x, coeffs_y, coeffs_z = curvaRegresionPolinomial3D(control_points, degree)
        curve_points = []
        for i in range(num_verts):
            t = i / (num_verts - 1)
            curve_points.append(curvaEvaluarPolinomio3D(coeffs_x, coeffs_y, coeffs_z, t))

    if abs(rotation_angle) > 0.001:
        curve_points = utilRotarPuntosAlrededorEje(curve_points, axis_start, axis_direction, rotation_angle)

    mesh_name = f"Curva{curve_type.title()}"
    mesh = bpy.data.meshes.new(mesh_name)
    obj = bpy.data.objects.new(f"{mesh_name}Obj", mesh)
    bpy.context.collection.objects.link(obj)
    edges = [(i, i + 1) for i in range(len(curve_points) - 1)]
    mesh.from_pydata(curve_points, edges, [])
    mesh.update()
    return obj


def elementGetPosicionPorModo(context, obj, bm, capture_mode):
    if context.mode != 'EDIT_MESH':
        return None, None

    if capture_mode == 'VERT':
        if bm.select_history and isinstance(bm.select_history[-1], bmesh.types.BMVert):
            elem = bm.select_history[-1]
            pos = obj.matrix_world @ elem.co
            key = (obj.name, 'VERT', elem.index)
            return pos, key
    elif capture_mode == 'EDGE':
        if bm.select_history and isinstance(bm.select_history[-1], bmesh.types.BMEdge):
            elem = bm.select_history[-1]
            mid = (elem.verts[0].co + elem.verts[1].co) / 2
            pos = obj.matrix_world @ mid
            key = (obj.name, 'EDGE', elem.index)
            return pos, key
    elif capture_mode == 'FACE':
        if bm.select_history and isinstance(bm.select_history[-1], bmesh.types.BMFace):
            elem = bm.select_history[-1]
            pos = obj.matrix_world @ elem.calc_center_median()
            key = (obj.name, 'FACE', elem.index)
            return pos, key

    return None, None


def objectoGetPosicionActiva(context):
    if context.mode != 'OBJECT':
        return None, None
    active_obj = context.active_object
    if not active_obj or active_obj.type != 'MESH':
        return None, None
    pos = active_obj.location.copy()
    key = (active_obj.name, 'OBJECT', 0)
    return pos, key


def capturarPosicionActiva(context, capture_mode):
    if capture_mode == 'OBJECT':
        return objectoGetPosicionActiva(context)

    obj = context.active_object
    if not obj or obj.type != 'MESH' or context.mode != 'EDIT_MESH':
        return None, None

    bm = bmesh.from_edit_mesh(obj.data)
    return elementGetPosicionPorModo(context, obj, bm, capture_mode)


def listaPuntosControlAgregar(lista_puntos, key, pos_global, max_puntos):
    obj_name, elem_type, elem_index = key

    is_new = not any(
        v['obj_name'] == obj_name and v['element_type'] == elem_type and v['index'] == elem_index
        for v in lista_puntos
    )

    if not is_new:
        return False

    if len(lista_puntos) >= max_puntos:
        lista_puntos.pop(0)

    lista_puntos.append({
        'obj_name': obj_name,
        'index': elem_index,
        'element_type': elem_type,
        'global_pos': pos_global.to_tuple(),
    })

    return True


def listaPuntosControlLimpiar(lista_puntos):
    lista_puntos.clear()
